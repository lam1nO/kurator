#include <iostream>

using namespace std;

int main()
{
    int N, a, flag = 1; // flag = 1, можем по дефолту решить, что число удовл.условию. а если какие то 2 цифры не удовлетворят, сразу поставить, что flag = 0.
    cin >> N;
    a = N / 10;
    while (a != 0){
        if (a % 10 <= N % 10){ // <= условие, с помощью которого определим что число не удовлетворяет
            flag = 0; // поэтому flag = 0. можно еще после этого сразу break написать.
        }
        N /= 10;
        a /= 10;
    }
    // cout << flag << endl; зачем это выводить?
    if (flag == 1){
        cout << "YES";
    }
    else
        cout << "NO";
    return 0;
}
