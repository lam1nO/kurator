C++.
1) int max = 99999; - чтобы определить максимум мы должны начальным значением брать заведомо маленькое число (а не наоборот)
   int max = -99999;

2) mass[max] - в данном виде это означает элемента массива mass с индексом max, а max - это переменная для хранения максимума.

3) цикл сдвига. во-1 - mass [n - i]. изначально i = 0, n - i = n, элемента с индексом n нет. во-2 нам нужно элементы после максимума
сдвинуть влево. то есть максимум будет последним. а тут во 1 mass[max]-непонятно что, во 2 нет сдвига именно после максимума.

4) нельзя называть элемент max, так как есть такая функция в c++. лучше назвать mx. 



#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int n, i, p;
    cin >> n;
    vector < int > mass(n);
    for (int i = 0; i < n; i++)
    {
        cin >> mass[i];
    }
    int max = -99999; // см. (1)
    for (int i = 0; i < n; i++)
    {
        if (max < mass[i]) // см. (2)
        {
            max = mass[i]; // см. (2)
        }
    }
    for (int i = 0; i < n; i++)
    {
	if (mass[i] == max){ // нашли max и двигаем его вправо.
	    swap(mass[i], mass[i + 1]); // в идеале mass[max (i + 1, n - 1)], но будет ошибка скорее всего из-за max(функция) и max(переменная)
	}
    }
    mass[n - 1] = 0;
    for (int i = 0; i < n; i++)
    {
        cout << mass[i] << " ";
    }
}
